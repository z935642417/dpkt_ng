# IE IDs
IE_HT_CAPA = 45; IE_HT_INFO = 61; IE_VHT_CAPA = 191; IE_VHT_OP = 192

class IEEE80211IEHTCapability(IEEE80211IE):
    """HT Capabilities (45). Fields: info(H) + ampdu(B) + mcs(16s) + ext_cap(H) + beam(I) + asel(B)."""
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 26:
            self.ht_cap_info = struct.unpack('<H', self.info[0:2])[0]
            self.ampdu_params = self.info[2]
            self.mcs_set = self.info[3:19]
            self.ht_ext_cap = struct.unpack('<H', self.info[19:21])[0]
            self.beamforming = struct.unpack('<I', self.info[21:25])[0]
            self.asel = self.info[25]

class IEEE80211IEHTInfo(IEEE80211IE):
    """HT Information (61)."""
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 22:
            self.primary_ch = self.info[0]
            self.ht_info_config = struct.unpack('<H', self.info[1:3])[0]
            self.basic_mcs = self.info[6:22]

class IEEE80211IEVHTCapability(IEEE80211IE):
    """VHT Capabilities (191)."""
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 12:
            self.vht_cap_info = struct.unpack('<I', self.info[0:4])[0]
            self.mcs_map = struct.unpack('<HH', self.info[4:8])
            # mcs_map is (rx_map, tx_map)

class IEEE80211IEVHTOperation(IEEE80211IE):
    """VHT Operation (192)."""
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 5:
            self.ch_width = self.info[0]
            self.ch1 = self.info[1]
            self.ch2 = struct.unpack('<H', self.info[2:4])[0]
            self.basic_mcs = struct.unpack('<H', self.info[4:6])[0]

for id_, cls in [(IE_HT_CAPA, IEEE80211IEHTCapability), (IE_HT_INFO, IEEE80211IEHTInfo),
                  (IE_VHT_CAPA, IEEE80211IEVHTCapability), (IE_VHT_OP, IEEE80211IEVHTOperation)]:
    register_ie(id_, cls)

### Task 2: RSN + ERP + Extended Cap + Country IEs

```python
IE_ERP = 42; IE_RSN = 48; IE_EXT_CAP = 127; IE_COUNTRY = 7

class IEEE80211IEERP(IEEE80211IE):
    def unpack(self, buf):
        super().unpack(buf)
        if self.info: self.flags = self.info[0]

class IEEE80211IERSN(IEEE80211IE):
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 8:
            self.version = struct.unpack('<H', self.info[0:2])[0]
            self.group_suite = self.info[2:6]
            self.pairwise_count = struct.unpack('<H', self.info[6:8])[0]
            off = 8
            self.pairwise = []
            for _ in range(self.pairwise_count):
                if off+4 <= len(self.info):
                    self.pairwise.append(self.info[off:off+4]); off += 4
            self.akm_count = struct.unpack('<H', self.info[off:off+2])[0]; off += 2
            self.akm = []
            for _ in range(self.akm_count):
                if off+4 <= len(self.info):
                    self.akm.append(self.info[off:off+4]); off += 4
            if off+2 <= len(self.info):
                self.caps = struct.unpack('<H', self.info[off:off+2])[0]

class IEEE80211IEExtCap(IEEE80211IE):
    def unpack(self, buf):
        super().unpack(buf); self.flags = self.info

class IEEE80211IECountry(IEEE80211IE):
    def unpack(self, buf):
        super().unpack(buf)
        self.country_code = self.info[:3] if len(self.info) >= 3 else b''
        self.triplets = []
        off = 3
        while off + 3 <= len(self.info):
            self.triplets.append({
                'channel': self.info[off], 'channels': self.info[off+1],
                'power': self.info[off+2] if isinstance(self.info[off+2], int) else struct.unpack('b', bytes([self.info[off+2]]))[0]
            }); off += 3

# Register all
for id_, cls in [(IE_ERP,IEEE80211IEERP),(IE_RSN,IEEE80211IERSN),(IE_EXT_CAP,IEEE80211IEExtCap),(IE_COUNTRY,IEEE80211IECountry)]:
    register_ie(id_, cls)
```

### Task 3: WMM IE + ProbeReq Body + ReassocReq Body

```python
class IEEE80211IEWMM(IEEE80211IE):
    """WMM/WME vendor IE (221, OUI 00:50:F2:02:01)."""
    def unpack(self, buf):
        super().unpack(buf)
        if len(self.info) >= 7 and self.info[0:4] == b'\x00\x50\xf2\x02':
            self.oui = self.info[0:3]; self.oui_type = self.info[3]
            self.subtype = self.info[4]; self.version = self.info[5]
            self.qos_info = self.info[6]
            self.params = []  # AC parameters
            off = 7
            while off + 4 <= len(self.info):
                aci = (self.info[off] >> 5) & 3; acm = (self.info[off] >> 4) & 1
                param = struct.unpack('<H', self.info[off+2:off+4])[0]
                self.params.append({'aci':aci,'acm':acm,'txop':param})
                off += 4
register_ie(221, IEEE80211IEWMM)
```
